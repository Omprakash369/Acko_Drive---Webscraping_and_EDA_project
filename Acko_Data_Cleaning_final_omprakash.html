def convert_price(price_str):
    price_str = str(price_str).strip()
    if 'L' in price_str:
        return float(price_str.replace('L', '').strip()) * 1e5
    elif 'Cr' in price_str:
        return float(price_str.replace('Cr', '').strip()) * 1e7
    else:
       
        try:
            return float(price_str)
        except ValueError:
            return np.nan
